24MessageReceiver::Result MessageReceiver::receiveMessage()
25{
26 char buffer[MessageReceiver::BUFFER_SIZE];
27
28 //recv 함수는 클라이언트 소켓의 수신 버퍼에서 데이터를 읽어옵니다.
29 // 인자 설명:.
30 // 첫 번째 인자: 데이터를 읽을 대상 소켓 (_clientSocket).
31 // 두 번째 인자: 수신 데이터를 저장할 버퍼 (buffer).
32 // 세 번째 인자: 최대 읽을 바이트 수 (BUFFER_SIZE - 1).
33 // 네 번째 인자: 플래그 (일반적으로 0 사용).
34 // MSG_PEEK : 데이터를 읽더라도 버퍼를 안지움 (헤더 확인 시 사용할 수 있음).
35 // MSG_OOB : 대역 외 데이터 수신.
36 // MSG_WAITALL : 전체 버퍼가 채워질 때까지 대기.
37 // 반환값인 receive_result는 다음을 의미합니다:.
38 // > 0: 실제로 읽은 바이트 수.
39 // == 0: 연결이 정상적으로 종료됨.
40 // < 0: 오류 발생 (예: 연결 끊김 또는 네트워크 오류).
41 int receive_result = recv(this->_clientSocket, buffer, BUFFER_SIZE - 1, 0);
42
43 // 수신이 제대로 이루어진 경우.
44 if (receive_result > 0)
45 {
46 buffer[receive_result] = '\0';
47 this->_lastMessage = std::string(buffer);
48
49 LOG_INFO("수신: " + this->_lastMessage);
50 // 수신 받은 문자열의 케리지 리턴을 제거합니다.
51 cleanMessage(this->_lastMessage);
52
53 // 수신받은 문자열이 quit인지를 체크합니다.
54 if (isQuitCommand(this->_lastMessage))
55 {
56 // 맞다면.
57 return (MessageReceiver::Result::CLIENT_QUIT);
58 }
59
60 return (MessageReceiver::Result::SUCCESS);
61 }
62 // 연결이 정상적으로 종료.
63 else if (receive_result == 0)
64 {
65 LOG_INFO("클라이언트가 연결을 종료했습니다.");
66 return (MessageReceiver::Result::CLIENT_DISCONNECTED);
67 }
68 // 오류 발생.
69 else
70 {
71 LOG_ERROR("메시지 수신을 실패했습니다.\n에러 코드: " + std::to_string(WSAGetLastError()));
72 return (MessageReceiver::Result::FAIL_RECEIVE);
73 }
74}

89{
