73void Program::runServerLoop()
74{
75 LOG_INFO("멀티클라이언트 서버 메인 루프를 시작합니다.");
76 LOG_INFO("최대 " + std::to_string(ClientManager::MAX_CLIENTS) + "명의 클라이언트가 동시 접속 가능합니다.");
77
78 // 멀티 서버 메인 루프 실행
79 MultiServer::Result result = this->_multiServer.runServerLoop();
80
81 // 결과에 따른 로그 출력
82 switch (result)
83 {
84 case MultiServer::Result::SUCCESS:
85 LOG_INFO("멀티클라이언트 서버가 정상적으로 종료되었습니다.");
86 break;
87 case MultiServer::Result::FAIL_LOOP:
88 LOG_ERROR("서버 루프 실행 중 오류가 발생했습니다.");
89 break;
90 case MultiServer::Result::SERVER_STOPPED:
91 LOG_INFO("서버가 사용자 요청에 의해 중지되었습니다.");
92 break;
93 default:
94 LOG_WARN("알 수 없는 서버 종료 상태입니다.");
95 break;
96 }
97}
