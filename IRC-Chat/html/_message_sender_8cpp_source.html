25MessageSender::Result MessageSender::broadcast(const std::string& message, SOCKET* sockets, int socket_count)
26{
27 // 전송할 소켓이 없는 경우.
28 if (socket_count == 0)
29 {
30 LOG_DEBUG("전송할 클라이언트가 없습니다.");
31 return (MessageSender::Result::TOTAL_FAIL);
32 }
33
34 // 클라이언트로 보낼 메세지로 포멧합니다.
35 std::string formatted_message = this->formatMessage(message);
36
37 // 전송 성공 수.
38 int success_count = 0;
39
40 for (int i = 0; i < socket_count; ++i)
41 {
42 if (this->sendMessage(formatted_message, sockets[i]))
43 {
44 success_count = success_count + 1;
45 }
46 }
47
48 // 결과에 따른 분기.
49 if (success_count == socket_count)
50 {
51 LOG_INFO("브로드캐스트 성공: " + message);
52 return (MessageSender::Result::SUCCESS);
53 }
54 else if (success_count > 0)
55 {
56 LOG_WARN("브로드캐스트 부분 실패: " + std::to_string(success_count) + "/" + std::to_string(socket_count));
57 return (MessageSender::Result::PARTIAL_FAIL);
58 }
59 else
60 {
61 LOG_ERROR("브로드캐스트 실패");
