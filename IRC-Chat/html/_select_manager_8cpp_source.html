58SelectManager::Result SelectManager::executeSelect(int timeout_sec)
59{
60 // 감시할 소켓이 없으면 바로 탈출합니다.
61 if (this->_socketCount == 0)
62 {
63 LOG_DEBUG("감시할 소켓이 없습니다.");
64 return (SelectManager::Result::NO_SOCKETS);
65 }
66
67 // fd_set을 복사합니다.
68 // select 함수가 인자의 fd_set을 수정하므로 원본을 수정하지 않기 위해 복사해서 사용합니다.
69 this->_copySet = this->_originSet;
70
71 // select에 사용할 timeout 설정.
72 timeval timeout = make_timer(timeout_sec, 0);
73
74 // Windows에서는 select함수의 첫번 째 매개변수가 무시됩니다.(원래는 최대 값의 소켓 id).
75 // 읽기 준비가 된 소켓을 수를 리턴합니다.
76 int result = select(0, &this->_copySet, nullptr, nullptr, &timeout);
77
78 // select 결과에 따른 분기.
79 if (result == SOCKET_ERROR)
80 {
81 LOG_ERROR("slect 함수 실행 실패\n 에러 코드: " + std::to_string(WSAGetLastError()));
82 return (SelectManager::Result::FAIL_SELECT);
83 }
84 else if (result == 0)
85 {
86 // 타임아웃 발생.
87 LOG_DEBUG("select 타임아웃 발생");
88 return (SelectManager::Result::TIMEOUT);
89 }
90 else
91 {
92 // 하나 이상의 소켓이 준비됨.
93 LOG_DEBUG("select 성공, 준비된 소켓 수: " + std::to_string(result));
94 return (SelectManager::Result::SUCCESS);
95 }
96}

98bool SelectManager::isSocketReady(SOCKET socket) const
99{
100 // FD_ISSET를 사용하여 fd_set에서 소켓이 읽기가 준비되었는지 체크합니다.
101 // select() 실행 후 복사된 fd_set에는 읽기가 준비된 소켓만 남아있습니다.
102 return (FD_ISSET(socket, &this->_copySet));
103}
