62 oss << "\tmaxSockets=" << this->_wsaData.iMaxSockets;
63 oss << ", maxUdpDg=" << this->_wsaData.iMaxUdpDg << '\n';
64 oss << "\tdesc=\"" << this->_wsaData.szDescription << "\"";
