26MultiServer::Result MultiServer::startServer()
27{
28 LOG_INFO("멀티클라이언트 서버를 시작합니다");
29
30 // TCP 소켓 생성
31 if (this->_tcpSocket.createTCPSocket() != TCPSocket::Result::SUCCESS)
32 {
33 return (MultiServer::Result::FAIL_START);
34 }
35
36 // 포트에 바인딩
37 if (this->_tcpSocket.bindTCPSocket(_port) != TCPSocket::Result::SUCCESS)
38 {
39 return (MultiServer::Result::FAIL_START);
40 }
41
42 // 리슨 시작
43 if (this->_tcpSocket.startListen() != TCPSocket::Result::SUCCESS)
44 {
45 return (MultiServer::Result::FAIL_START);
46 }
47
48 this->_isRunning = true;
49 LOG_INFO("멀티클라이언트 서버가 성공적으로 시작되었습니다");
50
51 return (MultiServer::Result::SUCCESS);
52}

54MultiServer::Result MultiServer::runServerLoop()
55{
56 LOG_INFO("서버 메인 루프를 시작합니다");
57
58 while (this->_isRunning)
59 {
60 // SelectManager를 사용해 fd_set 설정
61 this->_selectManager.setupFdSet();
62
63 // 서버 소켓 추가
64 this->_selectManager.addSocket(this->_tcpSocket.getSocket());
65
66 // 모든 클라이언트 소켓 추가
67 SOCKET client_sockets[ClientManager::MAX_CLIENTS];
68 int socket_count = _clientManager.getAllSockets(client_sockets, ClientManager::MAX_CLIENTS);
69
70 for (int i = 0; i < socket_count; ++i)
71 {
72 this->_selectManager.addSocket(client_sockets[i]);
73 }
74
75 // select 실행
76 SelectManager::Result select_result = this->_selectManager.executeSelect(1);
77
78 // select 결과 처리
79 switch (select_result)
80 {
81 case SelectManager::Result::SUCCESS:
82 // 활성 소켓 처리
83 break;
84
85 case SelectManager::Result::TIMEOUT:
86 // 타임아웃 - 다음 루프로 계속
87 continue;
88
89 case SelectManager::Result::FAIL_SELECT:
90 LOG_ERROR("select 실행 실패");
91 return (MultiServer::Result::FAIL_LOOP);
92
93 case SelectManager::Result::NO_SOCKETS:
94 // 소켓이 없으면 잠시 대기
95 Sleep(100);
96 continue;
97 }
98
99 // 서버 소켓 확인 (새로운 연결)
100 if (this->_selectManager.isSocketReady(this->_tcpSocket.getSocket()))
101 {
102 if (this->handleNewConnection() == false)
103 {
104 LOG_WARN("새로운 연결 처리 실패");
105 }
106 }
107
108 // 각 클라이언트 소켓 확인
109 for (int i = 0; i < ClientManager::MAX_CLIENTS; ++i)
110 {
111 SOCKET client_socket = this->_clientManager.getClientSocket(i);
112 if (client_socket != INVALID_SOCKET && this->_selectManager.isSocketReady(client_socket))
113 {
114 if (this->handleClientMessage(i) == false)
115 {
116 // 클라이언트 연결 종료
117 this->announceLeave(i);
118 this->_clientManager.removeClient(i);
119 }
120 }
121 }
122 }
123
124 LOG_INFO("서버 메인 루프가 종료되었습니다");
125 return (MultiServer::Result::SUCCESS);
126}

142bool MultiServer::handleNewConnection()
143{
144 // 새로운 클라이언트 연결 수락
145 SOCKET client_socket = this->_tcpSocket.acceptConnection();
146 if (client_socket == INVALID_SOCKET)
147 {
148 return (false);
149 }
150
151 // 클라이언트 추가
152 int client_index = this->_clientManager.addClient(client_socket);
153 if (client_index == -1)
154 {
155 // 최대 클라이언트 수 초과
156 std::string reject_message = "서버가 가득 찼습니다. 나중에 다시 시도해주세요.";
157 this->_messageSender.unicast(reject_message, client_socket);
158 closesocket(client_socket);
159 return (false);
160 }
161
162 // 환영 메시지 전송
163 this->sendWelcomeMessage(client_index);
164
165 // 다른 클라이언트들에게 참여 알림
166 this->announceJoin(client_index);
167
168 LOG_INFO("새로운 클라이언트 연결 완료 - 인덱스: " + std::to_string(client_index));
169 return (true);
170}

172bool MultiServer::handleClientMessage(int client_index)
173{
174 SOCKET client_socket = this->_clientManager.getClientSocket(client_index);
175 if (client_socket == INVALID_SOCKET)
176 {
177 return (false);
178 }
179
180 // MessageReceiver로 메시지 수신
181 MessageReceiver receiver(client_socket);
182 MessageReceiver::Result recv_result = receiver.receiveMessage();
183
184 switch (recv_result)
185 {
186 case MessageReceiver::Result::SUCCESS:
187 {
188 std::string message = receiver.getLastMessage();
189
190 // quit 명령 확인
191 if (receiver.isQuitCommand(message))
192 {
193 std::string goodbye_message = "[시스템] 안녕히 가세요!";
194 this->_messageSender.unicast(goodbye_message, client_socket);
195 return (false); // 연결 종료
196 }
197
198 // 클라이언트 닉네임.
199 std::string nickname = this->_clientManager.getClientNickname(client_index);
200 // 브로드캐스트 메시지 생성.
201 std::string broadcast_message = "[" + nickname + "]: " + message;
202
203
204 // 모든 클라이언트에게 브로드캐스트.
205 SOCKET all_sockets[ClientManager::MAX_CLIENTS];
206 int socket_count = this->_clientManager.getAllSockets(all_sockets, ClientManager::MAX_CLIENTS);
207 this->_messageSender.broadcast(broadcast_message, all_sockets, socket_count);
208
209 return (true);
210 }
211
212 case MessageReceiver::Result::CLIENT_DISCONNECTED:
213 LOG_INFO("클라이언트 연결 해제 - 인덱스: " + std::to_string(client_index));
214 return (false);
215
216 case MessageReceiver::Result::FAIL_RECEIVE:
217 LOG_ERROR("클라이언트 메시지 수신 실패 - 인덱스: " + std::to_string(client_index));
218 return (false);
219
220 default:
221 return (false);
222 }
223}

225void MultiServer::sendWelcomeMessage(int client_index)
226{
227 // 환영 메세지를 보낼 클라이언트 소켓을 가져옵니다.
228 SOCKET client_socket = this->_clientManager.getClientSocket(client_index);
229
230 // 소켓에 문제가 있다면.
231 if (client_socket == INVALID_SOCKET)
232 {
233 return ;
234 }
235
236 // 해당 클라이언트의 닉네임을 가져옵니다.
237 std::string nickname = this->_clientManager.getClientNickname(client_socket);
238 // 현재 접속 중인 유저의 수를 반환합니다.
239 int connectedClientCount = this->_clientManager.getConnectedClientCount();
240 // 환영 메세지를 생성합니다.
241 std::string welcome_message = makeWecomeMessage(nickname, connectedClientCount);
242
243 // 클라이언트에게 메세지를 전송합니다.
244 this->_messageSender.unicast(welcome_message, client_socket);
245}

247void MultiServer::announceJoin(int client_index)
248{
249 std::string nickname = this->_clientManager.getClientNickname(client_index);
250 // 클라이언트가 채팅방을 참여했다는 메세지 생성.
251 std::string join_message = "[시스템] " + nickname + "님이 채팅방에 참여했습니다.";
252
253 SOCKET all_sockets[ClientManager::MAX_CLIENTS];
254 int socket_count = this->_clientManager.getAllSockets(all_sockets, ClientManager::MAX_CLIENTS);
255
256 // 현재 채팅창에 들어온 클라이언트 소켓.
257 SOCKET new_client_socket = this->_clientManager.getClientSocket(client_index);
258
259 // 새로 들어온 클라이언트를 제외한 다른 클라이언트들에게 메세지 전송.
260 this->_messageSender.multicast(join_message, all_sockets, socket_count, new_client_socket);
261}

272

274 this->_messageSender.broadcast(leave_message, all_sockets, socket_count);
