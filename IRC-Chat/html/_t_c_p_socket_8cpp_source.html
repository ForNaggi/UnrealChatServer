24TCPSocket::Result TCPSocket::createTCPSocket()
25{
26 // socket함수를 사용해서 TCP 소켓을 생성합니다.
27 // AF_INET - IPv4를 의미.
28 // SOCK_STREAM - TCP 방식을 스트림 사용.
29 // IPPROTO_TCP - TCP 프로토콜 사용.
30 this->_tcpSocket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
31
32 // 소켓이 제대로 생성되었는지 확인.
33 if (this->_tcpSocket == INVALID_SOCKET)
34 {
35 LOG_ERROR("tcp소켓 생성을 실패했습니다.");
36 return (TCPSocket::Result::FAIL_CREATE);
37 }
38 LOG_INFO("tcp소켓 생성을 성공했습니다.");
39 return (TCPSocket::Result::SUCCESS);
40}

42TCPSocket::Result TCPSocket::bindTCPSocket(int port) const
43{
44 // 네트워크 주소 정보를 담는 구조체.
45 // sin_family - 주소 체계 (IPv4, IPv6). AF_INET은 IPv4를 의미.
46 // sin_addr.s_addr - 수신 IP, INADDR_ANY는 모든 IP에서 수신하겠다는 의미.
47 // sin_port - 포트 번호. htons - 네트워크 바이트 순서(빅엔디안)으로 변환해주는 함수.
48 sockaddr_in server_addr = {};
49 server_addr.sin_family = AF_INET;
50 server_addr.sin_addr.s_addr = INADDR_ANY;
51 server_addr.sin_port = htons(port);
52
53 // 소켓과 네트워크 주소를 바인드해줍니다.
54 // (sockaddr*)로 바꾸는 이유는 c언어가 다형성이 없기 때문입니다.
55 if (bind(this->_tcpSocket, (sockaddr*)&server_addr, sizeof(server_addr)) == SOCKET_ERROR)
56 {
57 LOG_ERROR("소켓 바인드 실패했습니다.\n에러코드: " + std::to_string(WSAGetLastError()));
58 return (TCPSocket::Result::FAIL_BIND);
59 }
60 LOG_INFO("소켓이 포트 " + std::to_string(port) + "에 바인드되었습니다.");
61 return (TCPSocket::Result::SUCCESS);
62}

64TCPSocket::Result TCPSocket::startListen()
65{
66 // 소켓을 수신 대기 상태로 전환해줍니다.
67 // 매개변수 두번 째(int backlog)는 최대 연결 대기 큐의 크기를 의미힙니다.
68 // 대기 큐가 크면 많은 클라이언트가 연결을 요청해도 대기한 뒤 연결할 수 있습니다.
69 // SOMAXCONN는 시스템에서 사용할 수 잇는 최대값입니다.
70 if (listen(this->_tcpSocket, SOMAXCONN) == SOCKET_ERROR)
71 {
72 LOG_ERROR("리슨 대기 시작이 실패했습니다.\n에러코드: " + std::to_string(WSAGetLastError()));
73 return (TCPSocket::Result::FAIL_LISTEN);
74 }
75
76 LOG_INFO("소켓이 클라이언트 연결을 대기하고 있습니다.");
77 return (TCPSocket::Result::SUCCESS);
78}

80SOCKET TCPSocket::acceptConnection()
81{
82 // 클라이언트와 연결될 소켓입니다.
83 SOCKET clientSocket = INVALID_SOCKET;
84
85 // 클라이언트 소켓에 대한 정보를 담을 구조체입니다.
86 sockaddr_in client_addr = {};
87 int addr_len = sizeof(client_addr);
88
89 // accept 함수 실행 시 대기 큐에 있던 클라이언트 요청을 하나 꺼내옵니다.
90 // accept 함수는 동기함수 임으로 클라이언트 연결까지 해당 함수에서 대기하게 됩니다.
91 // 소켓 구조체를 새로 생성하여 커널에 등록합니다. 내부 정보를 클라이언트 IP/Port 정보를 입력합니다.
92 // 소켓을 반환하고 반환된 소켓은 send()/recv()로 해당 클라이언트와 통신이 가능합니다.
93 clientSocket = accept(this->_tcpSocket, (sockaddr*)&client_addr, &addr_len);
94 if (clientSocket == INVALID_SOCKET)
95 {
96 LOG_ERROR("클라이언트와 연결에 실패했습니다.\n에러코드: " + std::to_string(WSAGetLastError()));
97 return (INVALID_SOCKET);
98 }
99
100 // 생성된 클라이언트 소켓에 대한 내부정보를 출력합니다.
101 this->logClientInfo(client_addr);
102 return (clientSocket);
103}

106{

110 // 열여있다면 소켓을 닫고 INVALID_SOCKET으로 초기화합니다.

127void TCPSocket::logClientInfo(const sockaddr_in& addr)
128{
129 // ip를 담을 배열.
130 char client_ip[INET_ADDRSTRLEN];
131
132 // inet_ntop는 바이트 배열을 리틀엔디안으로 변경합니다.
133 inet_ntop(AF_INET, &addr.sin_addr, client_ip, INET_ADDRSTRLEN);
134
135 // ntohs 역시 바이트 배열을 리틀엔디안으로 변경합니다.
136 std::string client_info = std::string(client_ip) + ":" + std::to_string(ntohs(addr.sin_port));
137 LOG_INFO("클라이언트가 연결되었습니다.\n네트워크 주소: " + client_info);
138}
