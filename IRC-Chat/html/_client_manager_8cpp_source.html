34int ClientManager::addClient(SOCKET client_socket)
35{
36 if (client_socket == INVALID_SOCKET)
37 {
38 LOG_WARN("유효하지 않은 소켓입니다.");
39 return (-1);
40 }
41
42 // 사용 가능한 인덱스 가져오기.
43 int index = this->getNextIndex();
44 if (index == -1)
45 {
46 LOG_WARN("클라이언트 추가 실패: 최대 클라이언트 수 초과");
47 return (-1);
48 }
49
50 // 클라이언트 소켓 저장.
51 this->_clientSockets[index] = client_socket;
52 this->_connectedSocketCount = this->_connectedSocketCount + 1;
53
54 LOG_INFO("클라이언트 추가 성공(player_" + std::to_string(index) + ")\n" + std::to_string(this->_connectedSocketCount) + "명");
55 return (index);
56}

58bool ClientManager::removeClient(int client_index)
59{
60 // 유효한 인덱스 범위인지를 체크.
61 if (this->isValidIndex(client_index) == false)
62 {
63 LOG_WARN("클라이언트 제거 실패: 유효하지 않은 인덱스 " + std::to_string(client_index));
64 return (false);
65 }
66
67 // 클라이언트가 연결되어 있는지 확인.
68 if (this->_clientSockets[client_index] == INVALID_SOCKET)
69 {
70 LOG_WARN("클라이언트 제거 실패: 이미 제거된 클라이언트 " + std::to_string(client_index));
71 return (false);
72 }
73
74 // 소켓 정리.
75 closesocket(this->_clientSockets[client_index]);
76 this->_clientSockets[client_index] = INVALID_SOCKET;
77 this->_connectedSocketCount = this->_connectedSocketCount - 1;
78
79 // 삭제한 소켓의 인덱스를 재사용 큐에 추가.
80 this->_availableList.push(client_index);
81
82 LOG_INFO("클라이언트 제거 완료(player_" + std::to_string(client_index) + ")\n남은 접속자: " + std::to_string(this->_connectedSocketCount) + "명");
83 return (true);
84}
